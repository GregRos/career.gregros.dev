    URL = require("url")
    function prettyUrl(arg)
        local replaced, count = string.gsub(arg, "mailto:", "")
            if count > 0 then
            tex.sprint(replaced)
            return
        end
        local url = URL.parse(string.lower(arg))
        tex.sprint(url.host .. url.path)
    end
\end{luacode}
\renewcommand{\url}[1]{%
    \href{#1}{\underline{\luadirect{prettyUrl("#1")}}}%
